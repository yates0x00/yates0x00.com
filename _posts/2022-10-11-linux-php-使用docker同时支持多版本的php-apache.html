傻瓜方式

更新apt

apt update

安装php

apt install php-common libapache2-mod-php php-cli

重启apache

# /etc/init.d/apache2 stop
# /etc/init.d/apache2 start

安装mysql, curl的php支持

# apt install php-mysql php-curl
